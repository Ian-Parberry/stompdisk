97DWORD Stomp(const uint64_t nWanted, uint64_t& nCreated, std::wstring& wstrFile){
98 const uint64_t nBytesInGB = 1073741824; //bytes in a gigabyte
99 const uint64_t nBytes = nBytesInGB*nWanted; //convert GB to bytes
100 LARGE_INTEGER nSize; //thanks for being weird, Microsoft
101 nSize.QuadPart = nBytes; //convert to bytes
102 DWORD dwErr = 0; //error code, initially no error
103 uint64_t nFileNum = 0; //number in file name
104
105 GetNextFile(nFileNum, wstrFile); //next file number and name
106
107 const HANDLE hFile = CreateFile(wstrFile.c_str(), GENERIC_WRITE,
108 FILE_SHARE_WRITE, NULL, CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
109 dwErr = GetLastError(); //check for error
110
111 if(SUCCEEDED(dwErr)){ //file has been opened, hFile is valid
112 SetFilePointerEx(hFile, nSize, 0, FILE_BEGIN); //set file ptr to start
113 dwErr = GetLastError(); //check for error
114
115 if(SUCCEEDED(dwErr)){ //no error
116 SetEndOfFile(hFile);
117 dwErr = GetLastError(); //check for error
118 } //if
119 GetFileSizeEx(hFile, &nSize); //read back file size
120 nCreated = nSize.QuadPart/nBytesInGB; //report back file size
121
122 CloseHandle(hFile); //close valid file handle
123 } //if
124
125 return dwErr; //error code, zero if no error
126} //Stomp
